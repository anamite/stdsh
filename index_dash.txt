const teamEvents = [
  { title: "Kick-off Meeting", date: "2025-04-22", time: "10:00 AM" },
  { title: "Design Review", date: "2025-04-23", time: "2:00 PM" },
  { title: "Product Launch", date: "2025-04-24", time: "4:00 PM" }

// Calendar logic (renders calendar, doesn't change)
(function(){
  const calendarContainer = document.getElementById("calendar-widget");
  if(!calendarContainer) return;

  let focusedDate = new Date();

  function formatDateISO(dt) {
    return dt.toISOString().slice(0,10);
  }

  function renderMiniCalendar(date) {
    const y = date.getFullYear(), m = date.getMonth();
    const today = new Date();
    const firstDay = new Date(y, m, 1);
    const lastDay = new Date(y, m + 1, 0);
    const startDay = firstDay.getDay();
    const daysInMonth = lastDay.getDate();
    const monthNames = [
      "January","February","March","April","May","June","July",
      "August","September","October","November","December"
    ];

    calendarContainer.innerHTML = `
      <div class="calendar-header">
        <button class="calendar-nav" id="prevMonth">&#8592;</button>
        <span id="calendarMonth">${monthNames[m]} ${y}</span>
        <button class="calendar-nav" id="nextMonth">&#8594;</button>
      </div>
      <div class="calendar-grid">
        <div class="calendar-day-name">Sun</div>
        <div class="calendar-day-name">Mon</div>
        <div class="calendar-day-name">Tue</div>
        <div class="calendar-day-name">Wed</div>
        <div class="calendar-day-name">Thu</div>
        <div class="calendar-day-name">Fri</div>
        <div class="calendar-day-name">Sat</div>
      </div>
    `;

    let grid = calendarContainer.querySelector('.calendar-grid');
    // Prev month
    let prevMonthEnd = new Date(y, m, 0).getDate();
    for(let i=0;i<startDay;i++) {
      let d = document.createElement('div');
      d.className = 'calendar-day other-month';
      d.textContent = prevMonthEnd - startDay + i + 1;
      grid.appendChild(d);
    }
    // This month
    for(let i=1;i<=daysInMonth;i++) {
      let d = document.createElement('div');
      d.className = 'calendar-day';
      d.textContent = i;
      let dayISO = `${y}-${String(m+1).padStart(2,'0')}-${String(i).padStart(2,'0')}`;
      const isToday =
        i === today.getDate() && m === today.getMonth() && y === today.getFullYear();
      if(isToday) d.classList.add('today');
      if(teamEvents.some(ev=>ev.date===dayISO)) {
        d.classList.add('has-event');
        d.title = "Events: " + teamEvents.filter(ev=>ev.date===dayISO).map(ev=>ev.title).join(', ');
      }
      grid.appendChild(d);
    }
    // Fill week
    let remain = (startDay+daysInMonth)%7;
    if(remain) {
      for(let i=1;i<=7-remain;i++) {
        let d=document.createElement('div');
        d.className='calendar-day other-month';
        d.textContent = i;
        grid.appendChild(d);
      }
    }
    // Nav
    calendarContainer.querySelector("#prevMonth").onclick=function(){
      focusedDate.setMonth(focusedDate.getMonth()-1);
      renderMiniCalendar(focusedDate);
    }
    calendarContainer.querySelector("#nextMonth").onclick=function(){
      focusedDate.setMonth(focusedDate.getMonth()+1);
      renderMiniCalendar(focusedDate);
    }
  }
  renderMiniCalendar(focusedDate);

  // Render events below
  function renderEventCards() {
    const eventsList = document.getElementById("calendar-events-list");
    if (!eventsList) return;
    eventsList.innerHTML = "";
    teamEvents.forEach(ev => {
      let card = document.createElement("div");
      card.className = "calendar-event-card";
      card.innerHTML = `
        <div class="calendar-event-title">${ev.title}</div>
        <div class="calendar-event-time">${ev.date} &nbsp; ${ev.time}</div>
      `;
      eventsList.appendChild(card);
    });
  }
  renderEventCards();
})();
